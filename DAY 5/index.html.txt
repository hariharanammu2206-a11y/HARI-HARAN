];

/* ===============================
   DATA PARSING (NO DATA LOST)
   =============================== */
const dates = goldData.map(d => d.Date);
const prices = goldData.map(d => parseFloat(d.Price.replace(/,/g,"")));
const volume = goldData.map(d => parseFloat(d["Vol."].replace("K","")));
const change = goldData.map(d => parseFloat(d["Change %"].replace("%","")));

/* ===============================
   KPI VALUES
   =============================== */

/* ===============================
   PRICE TREND
   =============================== */
new ApexCharts(document.querySelector("#priceChart"), {
  chart: { type: 'line', height: 320, zoom: { enabled: true } },
  series: [{ name: "Gold Price", data: prices }],
  xaxis: { categories: dates },
  stroke: { curve: 'smooth', width: 3 },
  theme: { mode: 'dark' },
  title: { text: "Gold Price Trend", align: "left" }
}).render();

/* ===============================
   VOLUME CHART
   =============================== */
new ApexCharts(document.querySelector("#volumeChart"), {
  chart: { type: 'bar', height: 260 },
  series: [{ name: "Volume (K)", data: volume }],
  xaxis: { categories: dates },
  theme: { mode: 'dark' },
  title: { text: "Trading Volume", align: "left" }
}).render();

/* ===============================
   CHANGE % CHART
   =============================== */
new ApexCharts(document.querySelector("#changeChart"), {
  chart: { type: 'bar', height: 260 },

  xaxis: { categories: dates },
  colors: ["#22c55e"],
  theme: { mode: 'dark' },
  title: { text: "Daily % Change", align: "left" }
